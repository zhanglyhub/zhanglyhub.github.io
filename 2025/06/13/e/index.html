经典的bfsbfsbfs状态信息的维护，我刚开始想成了dagdagdag拓扑排序生成所有最短路，显然这样是不行的，在任意一个位置我们要维护三个信息，pre,cur,nxtpre,cur,nxtpre,cur,nxt三个状态，所以的大体思路确定了使用bfsbfsbfs，那么如何实现呢？开始我维护了一个三维disdisdis数组，然而这样必然炸内存，在题目所给的这个数据量下，我们要开∏i=133000×4\prod_{i=1}^33000\times4∏i=13​3000×4字节大小的数组也就是大概100GB100GB100GB显然是不合理的，如何高效去维护三个状态呢？我们可以使用数组来维护pre−>curpre->curpre−>cur的路径信息，用set来维护cur−>nxtcur->nxtcur−>nxt的所有信息，这样我们不仅高效的减少了内存消耗，同时，在O(logN)O(logN)O(logN)的时间内实现快速查找。

#include<bits/stdc++.h>
#define inf 1 << 30
#define mem(a, b) memset(a, b, sizeof(a))
#define st first
#define ed second
using namespace std;


using vt = vector<int>;
using pii = pair<int, int>;
using vpi = vector<pii>;
using ll = long long;
const int N = 3005;
const int K = 1e5 + 5;
struct node {int last, cur, nxt, step;};
int n, m, k, dis[N][N], pa[N][N];
vt g[N];
int main() {
    ios::sync_with_stdio(0);
    cin.tie(0);
    cin >> n >> m >> k;
    for(int i = 1; i <= m; i++) {
        int a, b;
        cin >> a >> b;
        g[a].push_back(b);
        g[b].push_back(a);
    }
    vector<set<int>> p[N];
    for(int i = 0; i <= n; i++) p[i].resize(n + 1);
    for(int i = 1; i <= k; i++) {
        int a, b, c;
        cin >> a >> b >> c;
        p[a][b].insert(c);
    }
    for(int i = 1; i <= n; i++) {
        for(int j = 1; j <= n; j++) {
            dis[i][j] = inf;
            pa[i][j] = 0;
        }
    }
    queue<pii> q;
    for(auto v : g[1]) {
        q.push({v, 1});
        pa[v][1] = 0;
        dis[v][1] = 1;
    }
    //clog << "eror" << endl;
    int mn = inf;
    int pre_n = 0;
    while(!q.empty()) {
        int sz  = q.size();
        for(int i = 1; i <= sz; i++) {
            auto u = q.front();
            int cur = u.st;
            int pre = u.ed;
            q.pop();
            if(cur == n) {
                if(dis[cur][pre] < mn) {
                    mn = dis[cur][pre];
                    pre_n = pre;
                }
            }
            if(dis[cur][pre] > mn) continue;
            for(auto x : g[cur]) {
                if(p[pre][cur].count(x)) continue;
                if(dis[x][cur] > dis[cur][pre] + 1) {
                    dis[x][cur] = dis[cur][pre] + 1;
                    pa[x][cur] = pre;
                    q.push({x, cur});
                }
            }
        }
    }
    if(mn == inf) cout << -1 << endl;
    else {
        cout << mn << endl;
        vt path;
        int cur = n;
        int pre = pre_n;
        while(cur) {
            path.push_back(cur);
            if(cur == 1) break;
            int tmp = pa[cur][pre];
            cur = pre;
            pre = tmp;
        }
        reverse(path.begin(), path.end());
        for(int v : path) cout << v << ' ';
    }
    return 0;
}

这也是一道bfsbfsbfs状态类型题目，题目要求找到给定正整数nnn的最小倍数不能出现给定的整数mmm，像这类倍数问题，我们定义状态一般为当前数curmod  ncur \mod ncurmodn的余数是否为000如果为000我们就找到的一个数是nnn的倍数。假设当前数是curcurcur有:
curmod  n=pcur\mod n = pcurmodn=p

cur/n=acur / n = acur/n=a所以cur=a∗n+pcur = a * n + pcur=a∗n+p

我们尝试在curcurcur后加一个数ddd

有nxt=cur∗10+dnxt = cur * 10 + dnxt=cur∗10+d

而nxtp=(cur∗10+d)mod  nnxt_p = (cur * 10 + d) \mod nnxtp​=(cur∗10+d)modn

依据同余原理：
nxtp=(curp∗10+d)mod  nnxt_p = (cur_p * 10 +d) \mod nnxtp​=(curp​∗10+d)modn

所以只要我们知道当前状态的余数我们总能通过加ddd的方式来得到下一个余数，所以我们只需要用队列维护余数的状态即可。

#include <bits/stdc++.h>
#define inf 1 << 30
#define mem(a, b) memset(a, b, sizeof(a))
#define st first
#define ed second
using namespace std;


using ll = long long;
using ult = unordered_set<ll>;
using ust = unordered_set<int>;
using vt = vector<int>;
using pii = pair<int, int>;
using vpi = vector<pii>;
using ll = long long;
const int N = 10005;
ll n, m, cur, vis[N], pa[N], t; //因为要恢复路径，所以要vis标记一访问的余数
char dis[N]; //pa数组：我当前节点是由那一个节点拓展来的，dis数组记录路径
ult s; //注意bfs维护的是每一位数字，而层数就是当前来到的位数
int main() {
    for(t = 1; cin >> n >> m; t++) {
        mem(vis, 0);
        mem(dis, 0);
        mem(pa, -1);
        s.clear();
        for(int i = 1; i <= m; i++) {
            int a;
            cin >> a;
            s.insert(a);
        }
        queue<int> q;
            for(int d = 1; d <= 9; d++) {
                if(s.count(d)) continue;
                int c = d % n;
                if(!vis[c]) {
                vis[c] = 1;
                pa[c] = -2;//-2 表示虚拟父节点节点
                dis[c] = (char)(d + '0');
                q.push(c);
            }
        }
        bool flag = 0;
        string ans = "";
        while(!q.empty()) {
            int u = q.front();
            q.pop();
            if(u == 0) {
                flag = 1;
                int v = pa[u];
                string tmp = "";
                tmp += dis[u];
                ans += to_string(v);
                for(; v != -2; v = pa[v]) {
                    tmp += dis[v];
                    //cerr << dis[v] << ' ';
                }
                //ans += pa[v];
                //cerr << pa[v] << ' ' << ans << endl;
                reverse(tmp.begin(), tmp.end());
                ans = tmp;
                //cerr << ans;
                break;
            }
            for(int d = 0; d <= 9; d++) {
                if(s.count(d)) continue;
                int nxt = (u * 10 + d) % n;
                if(!vis[nxt]) {
                    vis[nxt] = 1;
                    pa[nxt] = u;
                    dis[nxt] = (char)(d + '0');
                    q.push(nxt);
                }
            }
        }
        if(flag) cout << "Case " << t << ": "<< ans << endl;
        else cout << "Case " << t << ": "<< -1 << endl;
    }
}

这是一种很妙的写法，如果我们暴力生成所有倍数，时间空间都会爆炸，而这种基于状态去枚举每一位余数的方法，则有效优化，时间复杂度O(N)O(N)O(N)，空间复杂度O(N)O(N)O(N)
