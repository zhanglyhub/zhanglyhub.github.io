    #include<bits/stdc++.h>
    #define inf 1 << 30
    using namespace std;
    
    class Solution {
    public:
        int shortestPathAllKeys(vector<string>& g) {
            int n = g.size(), m = g[0].size();
            int x0 = 0, y0 = 0, k = 0;
            int vis[31][31][1 << 7] = {};
            int dx[] = {0, 0, -1, 1};
            int dy[] = {-1, 1, 0, 0};
            struct node {int x, y, w, s;};
            queue<node> q;
            for(int i = 0; i < n; i++) {
                for(int j = 0; j < m; j++) {
                    if(g[i][j] == '@') {
                        x0 = i, y0 = j;
                    }
                    if(g[i][j] >= 'a' && g[i][j] <= 'z') {
                        k++;
                    }
                }
            }
            vis[x0][y0][0] = 1;
            q.push({x0, y0, 0, 0});
            while(!q.empty()) {
                auto[x, y, w, s] = q.front();
                q.pop();
                //cout << x << ' ' << y << ' ' << w <<' ' << s << endl;
                //cout << f(w) << ' ';
                if(f(w) == k) return s;
                for(int i = 0; i < 4; i++) {
                    int nx = x + dx[i];
                    int ny = y + dy[i];
                    if(nx >= 0 && nx < n && ny >= 0 && ny < m) {
                        if(g[nx][ny] == '#') continue;
                        if(g[nx][ny] >= 'a' && g[nx][ny] <= 'z') {
                            int nxt = 1 << (g[nx][ny] - 'a');
                            if(!vis[nx][ny][w | nxt]) {
                                vis[nx][ny][w | nxt] = 1;
                                q.push({nx, ny, w | nxt, s + 1});
                            }
                        }
                        if(g[nx][ny] >= 'A' && g[nx][ny] <= 'Z') {
                            int nxt = 1 << (g[nx][ny] - 'A');
                            if((w & nxt) && !vis[nx][ny][w]) {
                                vis[nx][ny][w] = 1;
                                q.push({nx, ny, w, s + 1});
                            } else continue;
                        }
                        if((g[nx][ny] == '.' || g[nx][ny] == '@') && !vis[nx][ny][w]) {
                            q.push({nx, ny, w, s + 1});
                            vis[nx][ny][w] = 1;
                        }
                    }//g[nx][ny] == '@'一定要特判，很坑
                }
            }
            // for(int i = 0; i < n; i++) {
            //     for(int j = 0; j < n; j++) {
            //         cout << dis[i][j];
            //     }
            //     cout << endl;
            // }
            return -1;
        }
        int f(int x) {
            int cnt = 0;
            while(x) {
                x &= (x - 1);
                cnt++;
            }
            return cnt;
        }
    };
